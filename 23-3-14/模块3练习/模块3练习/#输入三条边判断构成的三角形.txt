#输入三条边判断构成的三角形
print("请输入三角形的三边长");
a=int(input("请输入a="))
b=int(input("请输入b="))
c=int(input("请输入c="))
if (a+b>c and a+c>b and b+c>a):
    if(a==b and b==c):
        print("该三角形为等边三角形")
    elif(a==b or b==c or a==c):
        print("该三角形为等腰三角形")
    else:
        print("该三角形为其他三角形")
else:
    print("注意：不能构成合法三角形")
